COREx67SECRETME.hooknative

Hook Native ด้วย 64-bit Hash และกำหนด Callback เพื่อจัดการการดักจับ — Callback จะได้รับ ctx object พร้อม Helpers

x67SECRETME.hooknative(integer hash, function callback) → void
ParameterTypeDescription
hashnumberNative Hash (เช่น 0x7F8F1234)
callbackfunction(ctx)ถูกเรียกทุกครั้งที่ Native ทำงาน

-- Hook Native ด้วย Hash
x67SECRETME.hooknative(0xHASHVALUE, function(ctx)
    print("Native intercepted!")

    -- อ่าน Argument แรกเป็น Integer
    local arg0 = ctx:GetInt(0)
    print("Arg[0] = " .. arg0)

    return true
end)

MethodReturnDescription

-- เปลี่ยนพิกัดปลายทางก่อนเทเลพอร์ต
x67SECRETME.hooknative(0x12345678, function(ctx)
    local dest = ctx:GetVector3(0)

    -- เลื่อนพิกัดขึ้น 1.0
    dest.x = dest.x + 1.0
    dest.y = dest.y + 1.0

    ctx:SetVector3(0, dest)
    return true
end)

CTRLค่าที่ส่งกลับ (Return)

ค่าที่ return จาก Callback ควบคุมว่า Native ต้นฉบับจะทำงานหรือไม่

✓
return true
ให้ Native ทำงานต่อตามปกติ (อาจใช้ Arguments ที่แก้ไขแล้ว)
✗
return false
บล็อก Native — ไม่ให้ทำงานเลย

Override Return Value

MethodTypeDescription
ctx:SetReturnInt(value)integerOverride ค่า Return เป็น Int
ctx:SetReturnFloat(value)floatOverride ค่า Return เป็น Float
ctx:SetReturnBool(value)booleanOverride ค่า Return เป็น Bool
ctx:SetReturnVector3(vec)Vector3Override ค่า Return เป็น Vector3

-- ดักจับ GIVE_WEAPON_TO_PED แล้วแก้ไข Arguments
x67SECRETME.hooknative(0xBF0FD6E56C964FCB, function(ctx)
    local ped        = ctx:GetInt(0)
    local weaponHash = ctx:GetUInt(1)
    local ammo       = ctx:GetInt(2)

    print("[Hook] Weapon: " .. weaponHash .. " Ammo: " .. ammo)

    -- เพิ่มกระสุนเป็น 2 เท่า
    ctx:SetInt(2, ammo * 2)

    return true
end)
